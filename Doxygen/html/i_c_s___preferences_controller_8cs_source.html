488  //
489  // Helpers
490  //
491  public static Color GetTypeColor(Type type) {
492  Type t= iCS_Types.GetElementType(type);
493  if(t == typeof(bool)) return BoolTypeColor;
494  if(t == typeof(int)) return IntTypeColor;
495  if(t == typeof(float)) return FloatTypeColor;
496  if(t == typeof(string)) return StringTypeColor;
497  if(t == typeof(GameObject)) return GameObjectTypeColor;
498  if(t == typeof(Vector2)) return Vector2TypeColor;
499  if(t == typeof(Vector3)) return Vector3TypeColor;
500  if(t == typeof(Vector4)) return Vector4TypeColor;
501  return DefaultTypeColor;
502  }
